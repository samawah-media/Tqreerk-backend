# ── PyTorch (CUDA 12.1 wheels — match the Dockerfile base image) ─────────────
# Pin minor versions; the Hugging Face / Docling stack is sensitive to
# torch/transformers compatibility.
# Bumped 2.4.1 → 2.5.1 on 2026-05-12 so the Surya OCR swap can pick up
# surya-ocr 0.13.x (requires torch>=2.5.1). torchvision moves in lock-step
# (0.19.1 → 0.20.1). Both have cu121 cp311 wheels on the PyTorch index, so
# the Dockerfile base image (nvidia/cuda:12.1.0) doesn't need to change.
--extra-index-url https://download.pytorch.org/whl/cu121
torch==2.5.1
torchvision==0.20.1

# ── Document AI stack ────────────────────────────────────────────────────────
# Docling = layout detection + reading order + table structure (TableFormer)
# It bundles DocLayNet weights and TableFormer; no extra downloads.
# Bumped from 2.5.2 → 2.93.0 on 2026-05-08 to pick up the documented
# AcceleratorOptions API at docling.datamodel.accelerator_options. The
# old version had no public way to pin device=cuda, so layout silently
# fell back to CPU paths in some code branches and slowed ingest.
docling==2.93.0

# transformers / accelerate / timm / einops — required by docling 2.93's
# layout (RT-DETRv2) and table (TableFormer) models. Florence-2 was
# retired 2026-05-08 in favour of Vertex Gemini Vision (see
# pipeline/figures.py); these stay because docling-ibm-models pulls
# them transitively for its own models.
#   transformers 4.50: AutoModelForImageTextToText (docling chart pipeline)
#                       + rt_detr_v2 model type (docling layout model)
#   accelerate 1.4:    required by docling-ibm-models 3.13.x (>=1.2.1)
transformers==4.50.0
accelerate==1.4.0
einops==0.8.0
timm==1.0.11

# Surya OCR — Arabic + English OCR fallback for low-quality / scanned regions.
# Swapped from EasyOCR 2026-05-12: better Arabic line accuracy and multi-
# column / RTL handling. PyTorch-native; pulls weights via transformers
# into HF_HOME so it reuses our existing model cache. NOTE: GPL-3 /
# commercial dual license — production use above the free tier needs a
# commercial license from Datalab.
# 0.13.1 ships the class-based RecognitionPredictor / DetectionPredictor
# API; it requires torch>=2.5.1 (see torch pin above) and transformers
# >=4.41,<5 which our 4.50.0 pin satisfies.
surya-ocr==0.13.1

# sentence-transformers — multilingual text embedding (replaces Vertex AI text-embedding-004).
# `intfloat/multilingual-e5-base` is 768-dim (matches DB schema), ~1GB VRAM,
# strong on Arabic + English retrieval. Loads alongside Florence-2 with room
# to spare on the L4. Keeping all inference on the GPU service eliminates the
# Vertex network egress that was SSL-EOF'ing on Cloud Run.
sentence-transformers==3.2.1
